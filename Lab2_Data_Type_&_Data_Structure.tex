\begin{frame}[fragile]{Vectors}  % 注意：加入 fragile 選項很重要！
\texttt{c()} stands for concatenate, which combines elements into a vector
\begin{lstlisting}
score <- c(60, 75, 89, 95, 71, 90)
name <- c("John", "James", "Hank", "Jeff", "Ryan", "Ray")
class <- c(1, 1, 1, 2, 2, 2)
\end{lstlisting}

\texttt{rep()}: Replicate elements of vectors and lists
\begin{lstlisting}
rep(x = 1, times = 9)
rep(x = c(1, 2), each = 3) 
rep(x = c(1, 2, 3), times = c(2, 3, 5))
\end{lstlisting}

\texttt{seq()}: Tool used to generate regular sequences of numbers
\begin{lstlisting}
seq(from = 1, to = 5, by = 1)
seq(1, 9, by = 2)   
seq(0, 1, length.out = 11)
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Vectors}
What will these commands do?
\begin{lstlisting}
score[1]

\begin{frame}[fragile]{Matrices}
\begin{columns}

\column{0.7\textwidth}
\texttt{matrix(data, nrow, ncol, byrow)} creates a matrix
\begin{itemize}
    \item data: take a set of values
    \item nrow: specifies the number of rows
    \item ncol: specifies the number of columns
    \item byrow: indicates that the matrix is filled row-wise

\end{itemize}
\begin{lstlisting}
height <- c(175, 168, 180, 154, 185, 162, 170, 187, 149)
matrix(data = height, nrow = 3, ncol = 3, byrow = T)
matrix(data = height, nrow = 3, ncol = 3, byrow = F)
\end{lstlisting}
